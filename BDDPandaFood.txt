CREATE TABLE cliente (cedula_cliente int not null PRIMARY KEY, nombre varchar(20) not null, apellido varchar(20) not null,puntos int not null);

CREATE TABLE factura (codigo_factura int not null PRIMARY KEY, fecha date not null,precio_total int not null,cedula_cliente int not null,FOREIGN KEY (cedula_cliente) REFERENCES cliente (cedula_cliente));

CREATE TABLE detalle_factura(cedula_cliente int not null,codigo_factura int not null,  FOREIGN KEY (codigo_factura) REFERENCES factura (codigo_factura),FOREIGN KEY (cedula_cliente) REFERENCES cliente (cedula_cliente) )
